"カーソル下のファイルの存在をチェック
 function! g:check(str)
     let chk= getftype(a:str)
     if chk != ""
         echohl MoreMsg
         echo "### FOUND ### : "a:str
         echohl None
     else
         echohl ErrorMsg
         echo "### NOT FOUND ### : "a:str
         echohl None
     endif
 endfunction
 nnoremap [Space]j BvEy :call g:check("<C-r>"")<CR>
